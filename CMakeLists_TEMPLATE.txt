cmake_minimum_required(VERSION 3.10)
project(CS50Harvard C)

set(CMAKE_C_STANDARD 11)
set(SOURCE_EXE  initials.c)
set(SOURCE_LIB ${CMAKE_CURRENT_SOURCE_DIR}/Libs/libcs50-8.1.0/cs50.c)
set(CMAKE_INSTALL_PREFIX ${CMAKE_CURRENT_SOURCE_DIR}/Build/)

add_executable(build_1 ${SOURCE_EXE})

add_library(cs50 STATIC ${SOURCE_LIB})

target_link_libraries(build_1 cs50)

install(TARGETS build_1 DESTINATION ./)
